#include <stdio.h> 
int main()  
{
    char str[100], rev[100]; 
    int i, length = 0, vowels = 0, isPalindrome = 1;
    printf("Enter a string: ");  
    scanf("%s", str);
    while (str[length] != '\0')
    {
        length++;
    }
    for (i = 0; i < length; i++)
    {
        rev[i] = str[length - i - 1];
    }
    rev[length] = '\0'; 
    for (i = 0; i < length; i++)
    {
        char ch = str[i];
        if (ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u' ||
            ch == 'A' || ch == 'E' || ch == 'I' || ch == 'O' || ch == 'U')
        {
            vowels++;
        }
    }
    for (i = 0; i < length; i++)
    {
        if (str[i] != rev[i])
        {
            isPalindrome = 0;
            break;
        }
    }
    printf("\nReversed String: %s\n", rev);
    printf("Vowel Count: %d\n", vowels);
    if (isPalindrome)
        printf("The string is a Palindrome.\n");
    else
        printf("The string is Not a Palindrome.\n");
    return 0;
}

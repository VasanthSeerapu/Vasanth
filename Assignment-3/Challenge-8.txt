#include <stdio.h>
struct Student
{
    char name[50]; 
    int roll; 
    float marks; 
    char grade; 
};
int main() 
{
    int n, i;
    printf("Enter number of students: "); 
    scanf("%d", &n);
    struct Student s[n];
    for (i = 0; i < n; i++)
    {
        printf("\nStudent %d\n", i + 1);
        printf("Enter name: ");
        scanf("%s", s[i].name); 
        printf("Enter roll number: ");
        scanf("%d", &s[i].roll); 
        printf("Enter marks: ");
        scanf("%f", &s[i].marks);
        if (s[i].marks >= 90)
            s[i].grade = 'A';
        else if (s[i].marks >= 75)
            s[i].grade = 'B';
        else if (s[i].marks >= 60)
            s[i].grade = 'C';
        else if (s[i].marks >= 40)
            s[i].grade = 'D';
        else
            s[i].grade = 'F';
    }
    printf("\n--- Student Records ---\n");
    for (i = 0; i < n; i++)
    {
        printf("\nStudent %d Details\n", i + 1);
        printf("Name: %s\n", s[i].name);
        printf("Roll Number: %d\n", s[i].roll);
        printf("Marks: %.2f\n", s[i].marks);
        printf("Grade: %c\n", s[i].grade);
    }
    return 0;
}

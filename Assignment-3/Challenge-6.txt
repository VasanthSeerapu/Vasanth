#include <stdio.h> 
int main()  
{
    int n, i;
    float sum = 0, average;
    printf("Enter number of elements: ");
    scanf("%d", &n);
    int arr[n];
    printf("Enter %d elements:\n", n); 
    for (i = 0; i < n; i++) 
    {
        scanf("%d", &arr[i]);
        sum += arr[i];
    }
    int largest = arr[0];        
    int smallest = arr[0];  
    for (i = 1; i < n; i++) 
    {
        if (arr[i] > largest)
            largest = arr[i];

        if (arr[i] < smallest)
            smallest = arr[i];
    }
    average = sum / n;  
    printf("\nSum = %.2f\n", sum);
    printf("Average = %.2f\n", average);
    printf("Largest = %d\n", largest);
    printf("Smallest = %d\n", smallest);
    return 0; 
}
